def authenticate_ldap(username, password):
    ldap_server = "ldap.oci-iad-prd.marchex.com"  # Using the OpenLDAP host Tim referenced
    user_dn = f"uid={username},ou=users,dc=marchex,dc=com"

def check_user_access(target_user):
    """Checks all systems where the user has access."""
    timestamp = datetime.now().strftime("%Y-%m-%d %H:%M:%S")
    print(f"🔹 Checking access for {target_user} at {timestamp}")
    logging.info(f"Checking access for {target_user}")

    access_list = []

    if not os.path.exists(SYSTEMS_DIR):
        print("❌ Error: 'Systems/UIs' directory not found.")
        logging.error("Systems directory missing.")
        sys.exit(1)


    for script in os.listdir(SYSTEMS_DIR):
        if script.endswith(".py") and script.startswith("check_"):
            system_name = script.replace("check_", "").replace(".py", "").capitalize()
            script_path = os.path.join(SYSTEMS_DIR, script)

            try:
                cmd = ["python", script_path, target_user]
                process = subprocess.Popen(cmd, stdout=subprocess.PIPE, stderr=subprocess.PIPE, text=True, encoding="utf-8", bufsize=1)
                stdout_output, stderr_output = process.communicate()

                if stdout_output.strip():
                    print(f"✅ {system_name}: {stdout_output.strip()}")

                if stderr_output.strip():

                    logging.error(f"{system_name} Error: {stderr_output.strip()}")

            except Exception as e:
                print(f"❌ {system_name}: Error checking access - {e}")
                logging.error(f"Error checking access for {system_name}: {e}")

    return access_list

import logging
import psycopg2
import os
import sys
from get_secrets import get_secret


# ✅ Setup logging
logging.basicConfig(
    format="%(asctime)s - %(levelname)s - %(message)s",
    level=logging.INFO

DB_CONFIG = {
    "dbname": "adtrack_ro",
    "host": "vspool.som1.marchex.com",
    "port": 5432,
    "user": get_secret("_AdTrack_USER"),
    "password": get_secret("_AdTrack_PASS")
}



def get_db_connection():
    """Establish a database connection to AdTrack."""
    if not DB_CONFIG["user"] or not DB_CONFIG["password"]:
        logging.error("❌ Incorrect User or Password.")
        return None

    try:
        return psycopg2.connect(**DB_CONFIG)
    except Exception as e:
        logging.error(f"❌ Database connection failed: {e}")
        return None

def check_adtrack_access(user, vendor_email=None):
    """Check if a user has AdTrack UI access."""
    search_param = vendor_email or f"{user}@marchex.com"

    conn = get_db_connection()
    if not conn:
        error_msg = "❌ Database connection failed: Could not establish connection."
        logging.error(error_msg)
        print(error_msg)  # ✅ Ensure this prints so checkaccess.py captures it
        sys.stdout.flush()
        return None

    query = "SELECT email FROM at_user WHERE email ILIKE %s AND status != 'disabled'"
    try:
        with conn.cursor() as cur:
            cur.execute(query, (search_param,))
            result = cur.fetchall()

        conn.close()

        if result:
            output = f"✅ {user} has AdTrack UI access: {result}"
        else:
            output = f"❌ {user} does NOT have AdTrack UI access."

        logging.info(output)
        print(output)  # ✅ Ensure this prints so it's visible in checkaccess.py
        sys.stdout.flush()
        return result

    except Exception as e:
        error_msg = f"❌ Database query failed: {e}"
        logging.error(error_msg)
        print(error_msg)  # ✅ Ensure errors are visible
        sys.stdout.flush()
        return None

# ✅ GRANT Access Function
def grant_adtrack_access(user_email, real_name, is_staff=False):
    """Grants AdTrack UI access by inserting a user into the database."""
    conn = get_db_connection()
    if not conn:
        print("❌ Failed to connect to the AdTrack database.")
        return

    try:
        with conn.cursor() as cur:
            # Example insert — customize based on your DB schema
            cur.execute("""
                INSERT INTO at_user (email, name, status, staff_level)
                VALUES (%s, %s, 'active', %s)
                ON CONFLICT (email) DO UPDATE SET status = 'active'
            """, (user_email, real_name, 'staff' if is_staff else 'user'))
            conn.commit()

        print(f"✅ Granted access to {user_email} in AdTrack.")
        logging.info(f"✅ Granted access to {user_email} (staff={is_staff})")

    except Exception as e:
        logging.error(f"❌ Failed to grant access to {user_email}: {e}")
        print(f"❌ Grant failed for {user_email}: {e}")
    finally:
        conn.close()

# ✅ REMOVE Access Function
def remove_adtrack_access(user_email):
    """Disables an AdTrack user's account in the database."""
    conn = get_db_connection()
    if not conn:
        print("❌ Failed to connect to the AdTrack database.")
        return

    try:
        with conn.cursor() as cur:
            cur.execute("""
                UPDATE at_user
                SET status = 'disabled'
                WHERE email ILIKE %s
            """, (user_email,))
            conn.commit()

        print(f"✅ Removed access for {user_email} in AdTrack.")
        logging.info(f"✅ Disabled access for {user_email}.")

    except Exception as e:
        logging.error(f"❌ Failed to remove access for {user_email}: {e}")
        print(f"❌ Remove failed for {user_email}: {e}")
    finally:
        conn.close()

# ✅ Main Execution Block
def main():
    if len(sys.argv) < 3:
        print("Usage: python adtrack.py <check|grant|remove> <user> [real_name (if granting)] [--staff]")
        sys.exit(1)

    action = sys.argv[1].lower()
    user = sys.argv[2]
    real_name = sys.argv[3] if len(sys.argv) > 3 else None
    is_staff = "--staff" in sys.argv

    if action == "check":
        check_adtrack_access(user)
    elif action == "grant":
        if not real_name:
            print("❌ Error: Real name required to grant access.")
            sys.exit(1)
        grant_adtrack_access(user, real_name, is_staff)
    elif action == "remove":
        remove_adtrack_access(user)
    else:
        print("❌ Invalid action. Use 'check', 'grant', or 'remove'.")
        sys.exit(1)

if __name__ == "__main__":
    main()

def get_azure_token(mock_mode=False):
    """Fetch an Azure authentication token."""
    if mock_mode:
        logging.info("[MOCK] Skipping real token request.")
        return "mock_token"

    url = "https://login.microsoftonline.com/{}/oauth2/v2.0/token".format(AZURE_TENANT_ID)
    payload = {

        "client_secret": AZURE_CLIENT_SECRET,
        "scope": "https://graph.microsoft.com/.default",
        "grant_type": "client_credentials"
    }
    headers = {"Content-Type": "application/x-www-form-urlencoded"}

    try:
        response = requests.post(url, data=payload, headers=headers)
        response.raise_for_status()
        return response.json().get("access_token")
    except requests.exceptions.RequestException as e:
        logging.error(f"Failed to get Azure token: {e}")
        return None

def execute_access_action(action, ldap_user, ldap_pass, target_user, rt_ticket, mock_mode, log_area):
    """Runs access_management.py in a subprocess and updates GUI logs."""

    def run():
        try:
            args = ["python3", ACCESS_SCRIPT, action, target_user, "--ldap_user", ldap_user, "--ldap_pass", ldap_pass]

            if rt_ticket:
                args.extend(["--rt_ticket", rt_ticket])
            if mock_mode:
                args.append("--mock")


            process = subprocess.Popen(
                args,
                stdout=subprocess.PIPE,
                stderr=subprocess.PIPE,
                text=True,
                encoding="utf-8",
                bufsize=1
            )

            for line in iter(process.stdout.readline, ""):
                log_area.insert(tk.END, line.strip() + "\n")
                log_area.see(tk.END)

            for line in iter(process.stderr.readline, ""):
                log_area.insert(tk.END, f"⚠️ {line.strip()}\n")
                log_area.see(tk.END)

            process.stdout.close()
            process.stderr.close()
            process.wait()

        except Exception as e:
            log_area.insert(tk.END, f"❌ Error: {str(e)}\n", "error")
            logging.error(f"Error executing {ACCESS_SCRIPT}: {str(e)}")

    thread = threading.Thread(target=run, daemon=True)
    thread.start()

# ✅ Check user access
def check_user_access(target_user):
    """Checks all systems where the user has access."""
    timestamp = datetime.now().strftime("%Y-%m-%d %H:%M:%S")
    print(f"🔹 Checking access for {target_user} at {timestamp}")
    logging.info(f"Checking access for {target_user}")

    access_list = []

    found_any = False

    for directory in SYSTEMS_DIRS:
        if not os.path.exists(directory):
            continue  # silently skip missing dirs

        found_any = True

        for script in os.listdir(directory):
            if script.endswith(".py"):
                system_name = script.replace(".py", "").capitalize()
                script_path = os.path.join(directory, script)

                try:
                    cmd = [sys.executable, script_path, "check", target_user]
                    process = subprocess.Popen(cmd, stdout=subprocess.PIPE, stderr=subprocess.PIPE, text=True, encoding="utf-8", bufsize=1)
                    stdout_output, stderr_output = process.communicate()

                    if stdout_output.strip():
                        print(f"✅ {system_name}: {stdout_output.strip()}")
                        access_list.append(system_name)

                    if stderr_output.strip():
                        print(f"⚠️ {system_name} Error: {stderr_output.strip()}")
                        logging.error(f"{system_name} Error: {stderr_output.strip()}")

                except Exception as e:
                    print(f"❌ {system_name}: Error checking access - {e}")
                    logging.error(f"Error checking access for {system_name}: {e}")

    if not found_any:
        print("❌ Error: No valid system directories found (expected Systems/UIs or Systems/DBs).")
        logging.error("No system directories found.")

    return access_list


def find_script(system_name):
    for directory in SYSTEMS_DIRS:
        script_path = os.path.join(directory, f"{system_name.lower()}.py")
        if os.path.exists(script_path):
            return script_path
    return None

def authenticate_ldap(username, password):
    ldap_server = "ldap.oci-iad-prd.marchex.com"  # Using the OpenLDAP host Tim referenced
    user_dn = f"uid={username},ou=users,dc=marchex,dc=com"

    print(f"🔐 Trying LDAP bind as DN: {user_dn} to {ldap_server}", flush=True)

    try:
        server = Server(ldap_server, port=636, use_ssl=True, get_info=ALL)
        conn = Connection(
            server,
            user=user_dn,
            password=password,
            authentication=SIMPLE
        )

        if not conn.bind():
            print(f"❌ LDAP auth failed: {conn.result['description']}", flush=True)
            return False

        print(f"✅ Authenticated to LDAP as {user_dn}", flush=True)
        return True

    except Exception as e:
        print(f"❌ LDAP auth exception: {e}", flush=True)
        return False

# ✅ Check user access
def check_user_access(target_user):
    """Checks all systems where the user has access."""
    timestamp = datetime.now().strftime("%Y-%m-%d %H:%M:%S")
    print(f"🔹 Checking access for {target_user} at {timestamp}", flush=True)
    logging.info(f"Checking access for {target_user}")

    access_list = []

    found_any = False

    for directory in SYSTEMS_DIRS:
        if not os.path.exists(directory):
            continue  # silently skip missing dirs

        found_any = True

        for script in os.listdir(directory):
            if script.endswith(".py"):
                system_name = script.replace(".py", "").capitalize()
                script_path = os.path.join(directory, script)

                try:
                    cmd = [sys.executable, script_path, "check", target_user]
                    process = subprocess.Popen(
                        cmd,
                        stdout=subprocess.PIPE,
                        stderr=subprocess.PIPE,
                        text=True,
                        encoding="utf-8",
                        bufsize=1
                    )

                    stdout_output, stderr_output = process.communicate()

                    # Combine and normalize output
                    combined_output = (stdout_output or "") + (stderr_output or "")

                    # Basic success/failure logic
                    if "❌" in combined_output or "not have access" in combined_output.lower():
                        print(f"❌ {system_name}: {target_user} does not have access.", flush=True)
                    elif "✅" in combined_output or "has access" in combined_output.lower():
                        print(f"✅ {system_name}: {target_user} has access.", flush=True)
                        access_list.append(system_name)
                    else:
                        print(f"⚠️ {system_name}: Unable to determine access for {target_user}.", flush=True)

                    # Still log raw output for debugging
                    if stderr_output.strip():
                        logging.error(f"{system_name} STDERR: {stderr_output.strip()}")

                except Exception as e:
                    print(f"❌ {system_name}: Error checking access - {e}", flush=True)
                    logging.error(f"Error checking access for {system_name}: {e}")

    if not found_any:
        print("❌ Error: No valid system directories found (expected Systems/UIs or Systems/DBs).")
        logging.error("No system directories found.")

    elif not access_list:
        print(f"❌ {target_user} has no access to any known systems.", flush=True)

    return access_list

2	Manage Postgres Accounts
GeneralManage Postgres Accounts
Requirements
Appropriate Approvals, Postgres access.


Description
This is an SOP outlining the process that the SOC follows in order to create or remove user accounts on Marchex's postgres DBs. Access Granted through these commands is Read Only access. If the user requires WRITE access, get appropriate authorization and then pass the ticket to the DBA on-call.

For DBs <user> is always just the user alias, so "user" not "user@marchex.com". NOTE: for LDAP authenticating databases, this MUST match the user's LDAP exactly. If the user has a "-v" in their alias, you will need to put quotes around their alias in the query in order for the role to be created properly.

NOTE: If a database is not referenced in SOC documentation, consult the DBA team before attempting to grant access

NOTE: If you are seeing ERROR: cannot execute DROP (or CREATE) ROLE in a read-only transaction, that means you are working on a slave DB, you should switch to another DB ( this ONLY affects DB that are in AWS)

It is technically possible to automate access management for AdTrack UI using Python — but only if:

✅ You have a programmatic interface to work with, such as:
A REST API – If the AdTrack UI provides an API for user management, Python can interact with it via requests.
Direct DB write access – If AdTrack UI access is controlled via a backend database (e.g., at_user table), and you have write access and a documented schema, Python can manipulate it safely.
Web Automation – You could use tools like Selenium to mimic browser interactions (e.g., filling in forms to create users), but this is fragile and not recommended unless there's no API.
🚫 Current blockers based on your situation:
You’re using a read-only DB user – This prevents inserts/updates needed to automate granting/removing access.
There’s no known AdTrack UI API – All evidence suggests access is handled manually via the GUI (per SOP).
The AdTrack UI appears to be a web portal with no automation hooks.
🧠 Recommended Approach:
Split the automation:
adtrackDB.py: Automate check/remove (read-only) access to the database.
adtrackUI.py: Instead of automating, use this as an interactive wrapper that:
Logs the request.
Opens a browser to the right UI page.
Prompts the user to manually complete steps.
Tracks actions in the access_management.log for auditability.

This keeps the UI tool helpful without falsely implying automation is in place.
